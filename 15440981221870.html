shell脚本知识梳理（三） 流程控制

2018/12/6

if语句

这里的条件就是上节说的test命令的参数

if [ 条件 ]
then
    代码
fi

if [ 条件]
then
    代码
else
    代码
fi

if [ 条件 ]
then
    代码
elif [ 条件 ]
then
    代码
fi

for语句

for 变量名 in item1 item2 item3
do
    代码    
done

for name in "rao" "jun" "bo"
do
    echo ${name}
done    

//列出当前目录所有文件(用的for in)
file=`pwd`/*;
for name in ${file}
do
    echo ${name}
done

//for 的c形式
for (( i = 0; i < 10; i++ )); do
    echo $i;
    for (( j = 0; j < 10; j++ )); do
        echo $j;
    done
done

//一个一个显示文件列表
for i in `ls `
do
echo "这是文件${i}";
done

while

while [] 
do

done

//示例
var=10;
while [ ${var} -gt 0 ];
do
echo $var;
var=`expr ${var} - 1`;
done

switch case

number=1
case $number in
    1) echo "等于1";;
    2) echo "等于2";;
esac
